% -----------------------------------------------------------------
% TABLE OF CONTENTS, LIST OF FIGURES, BIB, etc. FORMAT

\renewcommand*\l@section{\@dottedtocline{1}{0em}{1.5em}} % nums are: {toc_level}{number_indent}{number_width}

\renewcommand*\l@subsubsection{\@dottedtocline{3}{3.75em}{3.0em}}
